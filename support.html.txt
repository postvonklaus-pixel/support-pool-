const uniqueChars = ['\u200B','\u200C','\u200D','\uFEFF','\u2060','\u2061','\u2062','\u2063'];
function makeUnique(text) {
    let u = text;
    for (let i = 0; i < 3; i++) u += uniqueChars[Math.floor(Math.random() * uniqueChars.length)];
    return u;
}

function handleLogin() {
    const input = document.getElementById('loginInput');
    const username = input.value.trim().toLowerCase();

    if (!username) { showToast('❌ Bitte Namen eingeben!'); return; }
    input.value = '';

    if (!appData.members) appData.members = {};
    if (!appData.groups) appData.groups = {};
    if (!appData.videos) appData.videos = {};

    if (appData.members[username]) {
        const member = appData.members[username];
        if (member.status === 'inactive') {
            const tg = findGroupForExistingMember();
            if (tg) {
                member.status = 'active';
                member.group = tg.id;
                tg.members.push(username);
                showToast('✨ Willkommen zurück! Du wurdest reaktiviert.');
            } else {
                showToast('❌ Keine Gruppe verfügbar!');
                return;
            }
        }
        appData.currentUser = username;
        showToast('🌟 Willkommen zurück, ' + username + '!');
    } else {
        const tg = findGroupForNewMember();
        if (!tg) { showToast('❌ Keine Gruppe verfügbar!'); return; }

        appData.members[username] = {
            name: username,
            role: 'member',
            group: tg.id,
            status: 'active',
            stats: {
                given: 0,
                received: 0,
                balance: 0,
                joined: Date.now()
            },
            myVideos: [],
            lastPostDate: null,
            newbiePostsLeft: 3,
            rewards: ['newbie']
        };
        tg.members.push(username);
        appData.currentUser = username;
        showToast('🎉 Herzlich willkommen in ' + tg.name + '!');
    }

    document.getElementById('loginSection').style.display = 'none';
    document.getElementById('mainPanel').style.display = 'block';
    document.getElementById('userWelcome').style.display = 'flex';
    document.getElementById('currentUserNameDisplay').textContent = appData.currentUser;

    const user = appData.members[appData.currentUser];
    document.getElementById('userRoleBadge').textContent = user.role === 'admin' ? 'Admin' : 'Member';

    if (user.role === 'admin') {
        document.getElementById('adminPanel').style.display = 'block';
        renderAdminGroups();
    }

    renderDashboard();
    checkExpiredVideos();
    saveData();
}

function findGroupForNewMember() {
    const sorted = Object.values(appData.groups).sort((a,b) => a.number - b.number);
    for (let g of sorted) { if (g.members.length < 10) return g; }

    const num = appData.settings.nextGroupId;
    const id = 'gruppe' + num;
    appData.groups[id] = {
        id: id,
        name: 'SUPPORT POOL ' + num,
        number: num,
        members: [],
        created: Date.now()
    };
    appData.settings.nextGroupId = num + 1;
    return appData.groups[id];
}

function findGroupForExistingMember() {
    const sorted = Object.values(appData.groups).sort((a,b) => a.number - b.number);
    for (let g of sorted) { if (g.members.length < 10) return g; }
    return null;
}

function showPostModal() {
    const user = appData.members[appData.currentUser];
    const cp = checkCanPost(user);
    if (!cp.allowed) { showToast(cp.message); return; }
    document.getElementById('modalVideoUrl').value = '';
    document.getElementById('postVideoModal').style.display = 'flex';
    setTimeout(() => document.getElementById('modalVideoUrl').focus(), 100);
}

function checkCanPost(user) {
    if (user.newbiePostsLeft > 0)
        return { allowed: true, message: `🎁 Willkommensgeschenk: Noch ${user.newbiePostsLeft} Freiposts` };
    if (user.stats.balance <= 0)
        return { allowed: false, message: '💫 Erst supporten, dann posten! Aktuell: ' + user.stats.balance };
    if (user.lastPostDate && (Date.now() - user.lastPostDate) < 86400000) {
        const h = Math.ceil((86400000 - (Date.now() - user.lastPostDate)) / 3600000);
        return { allowed: false, message: `⏳ Bitte warte noch ${h}h` };
    }
    return { allowed: true, message: '✅ Du kannst ein Video posten' };
}

function closePostModal() { document.getElementById('postVideoModal').style.display = 'none'; }

function postVideo() {
    const url = document.getElementById('modalVideoUrl').value.trim();
    if (!url) { showToast('❌ Bitte URL eingeben!'); return; }

    const user = appData.members[appData.currentUser];
    const videoId = 'video_' + Date.now() + '_' + Math.floor(Math.random()*10000);
    const group = appData.groups[user.group];
    const groupNumber = group ? group.number : 1;
    const now = new Date();

    let comment = appData.settings.commentTemplate
        .replace(/{gruppe}/g, groupNumber)
        .replace(/{name}/g, user.name)
        .replace(/{member}/g, appData.currentUser)
        .replace(/{date}/g, now.toLocaleDateString('de-DE'))
        .replace(/{time}/g, now.toLocaleTimeString('de-DE', { hour:'2-digit', minute:'2-digit' }))
        .replace(/\\n/g, '\n');

    appData.videos[videoId] = {
        id: videoId,
        owner: user.name,
        group: user.group,
        groupNumber: groupNumber,
        url: url,
        postedAt: Date.now(),
        requiredSupports: 10,
        currentSupports: 0,
        supporters: [],
        comment: comment,
        status: 'waiting',
        expiresAt: Date.now() + 86400000
    };
    user.myVideos.push(videoId);
    user.lastPostDate = Date.now();
    if (user.newbiePostsLeft > 0) user.newbiePostsLeft--;

    closePostModal();
    renderDashboard();
    showToast('✅ Video erfolgreich gepostet!');
    saveData();
}

    const video = appData.videos[videoId];
    const videoOwner = appData.members[video.owner];

    if (user.stats.balance < 0 && user.newbiePostsLeft === 0) { showToast('❌ Kein Guthaben!'); return false; }
    if (video.supporters.includes(user.name)) { showToast('❌ Bereits supportet!'); return false; }
    if (video.status !== 'waiting') { showToast('❌ Nicht mehr verfügbar!'); return false; }

    user.stats.given++;
    user.stats.balance = user.stats.given - user.stats.received;
    if (videoOwner) {
        videoOwner.stats.received++;
        videoOwner.stats.balance = videoOwner.stats.given - videoOwner.stats.received;
    }
    video.currentSupports++;
    video.supporters.push(user.name);

    navigator.clipboard.writeText(makeUnique(video.comment)).catch(()=>{});
    showToast('💫 Premium Kommentar kopiert!');
    if (video.currentSupports >= video.requiredSupports) {
        video.status = 'completed';
        showToast('✨ Video von @' + video.owner + ' ist komplett!');
    }
    renderDashboard();
    saveData();
    return true;
}

function removeMember(groupId, memberName) {
    const group = appData.groups[groupId];
    if (!group) return;
    group.members = group.members.filter(m => m !== memberName);
    if (appData.members[memberName]) {
        appData.members[memberName].status = 'inactive';
        appData.members[memberName].group = null;
    }
    showToast('✅ ' + memberName + ' wurde entfernt');
    renderAdminGroups();
    saveData();
}

function cleanupInactiveMembers() {
    const now = Date.now();
    const thr = appData.settings.inactiveDays * 86400000;
    let removed = 0;
    Object.values(appData.groups).forEach(group => {
        const active = [];
        group.members.forEach(mn => {
            const m = appData.members[mn];
            if (!m) return;
            const last = m.lastPostDate || 0;
            if (last > 0 && (now - last) > thr && m.role !== 'admin') {
                m.status = 'inactive';
                removed++;
            } else {
                active.push(mn);
            }
        });
        group.members = active;
    });
    showToast(removed > 0 ? `✨ ${removed} inaktive Mitglieder bereinigt` : '✅ Keine inaktiven Mitglieder');
    if (removed > 0) { renderAdminGroups(); saveData(); }
}

function renderOpenVideos(user) {
    const container = document.getElementById('openVideosList');
    const open = Object.values(appData.videos).filter(v =>
        v.status === 'waiting' && v.owner !== user.name && !v.supporters.includes(user.name)
    );
    document.getElementById('openVideosCount').textContent = open.length;

    if (!open.length) {
        container.innerHTML = '<div style="color:var(--text-tertiary);text-align:center;padding:30px;">✨ Aktuell keine offenen Videos</div>';
        return;
    }
    open.sort((a,b) => (a.requiredSupports-a.currentSupports)-(b.requiredSupports-b.currentSupports));
    container.innerHTML = open.map(v => {
        const left = v.expiresAt - Date.now();
        const h = Math.max(0, Math.floor(left/3600000));
        const m = Math.max(0, Math.floor((left%3600000)/60000));
        const urgent = left < 7200000;
        const dis = user.stats.balance <= 0 && user.newbiePostsLeft === 0;
        return `
        <div class="video-item waiting ${urgent ? 'expiring' : ''}">
            <div class="video-header">
                <span class="video-owner">@${v.owner}</span>
                <span class="video-status">${v.requiredSupports-v.currentSupports} nötig</span>
            </div>
            <div class="video-progress">
                <div class="progress-bar"><div class="progress-fill" style="width:${(v.currentSupports/v.requiredSupports)*100}%"></div></div>
                <span class="progress-text">${v.currentSupports}/${v.requiredSupports}</span>
            </div>
            <div class="video-timer ${urgent?'warning':''}">⏰ Noch ${h}h ${m}m</div>
            <button class="support-btn" onclick="openVideoAndSupport('${v.id}','${v.url}')" ${dis?'disabled':''}>
                ✨ Jetzt supporten & öffnen
            </button>
        </div>`;
    }).join('');
}

function renderMyVideos(user) {
    const container = document.getElementById('myVideosList');
    document.getElementById('myVideosCount').textContent = user.myVideos.length;
    if (!user.myVideos.length) {
        container.innerHTML = '<div style="color:var(--text-tertiary);text-align:center;padding:30px;">📭 Noch keine Videos gepostet</div>';
        return;
    }
    const last = [...user.myVideos].map(id => appData.videos[id]).filter(v=>v)
        .sort((a,b) => b.postedAt-a.postedAt).slice(0,3);
    container.innerHTML = last.map(v => {
        const icon = v.status==='completed'?'✅':v.status==='expired'?'⏰':'⏳';
        return `<div style="display:flex;justify-content:space-between;padding:10px 0;border-bottom:1px solid var(--glass-border);">
            <span style="color:var(--text-primary);">${icon} ${v.currentSupports}/${v.requiredSupports}</span>
            <span style="color:var(--text-tertiary);">${new Date(v.postedAt).toLocaleDateString('de-DE')}</span>
        </div>`;
    }).join('');
}

function checkExpiredVideos() {
    let changed = false;
    Object.values(appData.videos || {}).forEach(v => {
        if (v.status === 'waiting' && v.expiresAt < Date.now()) { v.status = 'expired'; changed = true; }
    });
    if (changed && appData.currentUser) { renderDashboard(); saveData(); }
}

    document.getElementById('importModal').style.display = 'flex';

function processImport() {
    const raw = document.getElementById('importData').value.trim();
    if (!raw) { showToast('❌ Bitte Daten einfügen!'); return; }
    try {
        const imp = JSON.parse(raw);
        if (!imp.groups || !imp.members || !imp.videos) {
            showToast('❌ Ungültiges Format! groups / members / videos fehlen.'); return;
        }
        const preservedUser = appData.currentUser;
        appData.groups = imp.groups;
        appData.members = imp.members;
        appData.videos = imp.videos;
        appData.settings = imp.settings || appData.settings;
        appData.currentUser = preservedUser;
        appData.version = 'premium_v1';

        saveData();
        closeImportModal();
        showToast('✅ Daten erfolgreich importiert!');

        if (appData.currentUser && appData.members[appData.currentUser]) {
            renderDashboard();
            if (appData.members[appData.currentUser].role === 'admin') renderAdminGroups();
        } else {
            logout();
        }
    } catch(e) {
        showToast('❌ JSON-Fehler: ' + e.message);
    }
}
